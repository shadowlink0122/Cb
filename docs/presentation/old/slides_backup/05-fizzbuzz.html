FizzBuzz - Cb固有の書き方

interface FizzBuzzable {
    string fizzbuzz();
}

struct Number {
    int value;
}

impl FizzBuzzable for Number {
    string fizzbuzz() {
        if (self.value % 15 == 0) {
            return "FizzBuzz";
        } else if (self.value % 3 == 0) {
            return "Fizz";
        } else if (self.value % 5 == 0) {
            return "Buzz";
        }
        return string(self.value);
    }
}

void main() {
    for (int i = 1; i <= 100; i++) {
        Number n;
        n.value = i;
        println(n.fizzbuzz());
    }
}

interface を使うことで、構造体に振る舞いを追加できます。
