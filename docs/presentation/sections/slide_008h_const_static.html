型修飾子 - const/static

const修飾子の2つの用途

// 1. 変数の再代入を禁止
const int MAX_SIZE = 100;
// MAX_SIZE = 200;  // エラー！

// 2. ポインタの変更を禁止
int value = 42;
const int* ptr1 = &value;  // 値の変更不可
// *ptr1 = 100;  // エラー！

int* const ptr2 = &value;  // ポインタの変更不可
// ptr2 = nullptr;  // エラー！

// 両方を禁止
const int const ptr3 = &value;

static修飾子

// staticローカル変数（状態を保持）
int counter() {
    static int count = 0;
    count = count + 1;
    return count;
}

void main() {
    println(counter());  // 1
    println(counter());  // 2
    println(counter());  // 3
}

ポイント :
• const: 変数やポインタの変更を防ぐ
• static: 関数呼び出し間で状態を保持
