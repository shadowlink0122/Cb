Interface/Impl - Goのシンプルさ

  // Interfaceの定義（Go風）
interface Shape {
    int area();
    void draw();
}
struct Rectangle {
    int width;
    int height;
};
// Implで実装（Rust風）
impl Shape for Rectangle {
    int area() {
        return self.width * self.height;
    }
    void draw() {
        println("Drawing rectangle: {self.width}x{self.height}");
    }
}
